# 常用模式与惯用法

## 概述

Go 有自己的编码风格和惯用模式。从 Java/C 转 Go 的人最容易犯错的地方不是语法，而是"写的不是 Go"。

## options 模式（函数式选项）

适合有大量可选参数的构造：

```go
type Client struct {
    timeout  time.Duration
    retries  int
    endpoint string
    logger   *log.Logger
}

type Option func(*Client)

func WithTimeout(t time.Duration) Option {
    return func(c *Client) { c.timeout = t }
}

func WithRetries(r int) Option {
    return func(c *Client) { c.retries = r }
}

func WithLogger(l *log.Logger) Option {
    return func(c *Client) { c.logger = l }
}

func NewClient(endpoint string, opts ...Option) *Client {
    c := &Client{
        endpoint: endpoint,
        timeout:  30 * time.Second,   // 默认值
        retries:  3,
    }
    for _, opt := range opts {
        opt(c)
    }
    return c
}

// 使用
client := NewClient("https://api.example.com",
    WithTimeout(10*time.Second),
    WithLogger(log.Default()),
)
```

## Builder 模式

适合逐步构造复杂对象：

```go
type ImageBuilder struct {
    path    string
    width   int
    height  int
    quality int
    format  string
}

func NewImageBuilder(path string) *ImageBuilder {
    return &ImageBuilder{
        path:    path,
        quality: 85,
        format:  "jpeg",
    }
}

func (b *ImageBuilder) Resize(w, h int) *ImageBuilder {
    b.width, b.height = w, h
    return b
}

func (b *ImageBuilder) Quality(q int) *ImageBuilder {
    b.quality = q
    return b
}

func (b *ImageBuilder) Format(f string) *ImageBuilder {
    b.format = f
    return b
}

func (b *ImageBuilder) Save() error {
    // 处理并保存
    return nil
}

// 使用
NewImageBuilder("photo.jpg").
    Resize(800, 600).
    Quality(90).
    Format("png").
    Save()
```

## 单例模式（线程安全）

```go
var (
    instance *Database
    once     sync.Once
)

func GetDatabase() *Database {
    once.Do(func() {
        instance = connectToDB()
    })
    return instance
}
```

## 工厂模式

```go
type ImageProcessor interface {
    Process(path string) (*Image, error)
}

type JpegProcessor struct{}
type PngProcessor struct{}

func NewProcessor(format string) ImageProcessor {
    switch format {
    case "jpg", "jpeg":
        return &JpegProcessor{}
    case "png":
        return &PngProcessor{}
    default:
        return nil
    }
}
```

## 健康检查模式

```go
type HealthCheckable interface {
    Health() error
}

type App struct {
    db      *sql.DB
    cache   *redis.Client
}

func (a *App) Health() map[string]string {
    result := make(map[string]string)

    if err := a.db.Ping(); err != nil {
        result["database"] = fmt.Sprintf("DOWN: %v", err)
    } else {
        result["database"] = "UP"
    }

    if err := a.cache.Ping(); err != nil {
        result["cache"] = fmt.Sprintf("DOWN: %v", err)
    } else {
        result["cache"] = "UP"
    }

    return result
}
```

## 上下文传递

```go
type contextKey string

const UserIDKey contextKey = "user_id"

func WithUserID(ctx context.Context, id int64) context.Context {
    return context.WithValue(ctx, UserIDKey, id)
}

func GetUserID(ctx context.Context) (int64, bool) {
    id, ok := ctx.Value(UserIDKey).(int64)
    return id, ok
}
```

## 重试模式

```go
func Retry(attempts int, sleep time.Duration, fn func() error) error {
    var err error
    for i := 0; i < attempts; i++ {
        err = fn()
        if err == nil {
            return nil
        }
        if i < attempts-1 {
            time.Sleep(sleep)
        }
    }
    return fmt.Errorf("after %d attempts: %w", attempts, err)
}
```

## Go 编码风格

### 命名

```go
// 缩写全大写
var HTTPClient *http.Client  // ✅
var httpClient *http.Client  // ❌

// 单字母变量只在短作用域
for i := 0; i < 10; i++ {}   // ✅

// 不写 getter 前缀
user.Name()         // ✅
user.GetName()      // ❌

// setter 用 Set 前缀
func (u *User) SetName(name string) { u.name = name }
```

### 包名

```go
package image  // ✅ 目录名就是包名
package imagemanager  // ❌ 太长了
```

### error 处理

```go
if err != nil {
    return err  // ✅ 标准模式
}
```
